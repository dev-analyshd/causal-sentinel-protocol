sdk/python/src/csp_sdk.py
"""Python SDK for Causal Sentinel Protocol"""
from typing import Optional, Dict, Any
import requests
import asyncio
from dataclasses import dataclass

@dataclass
class CoherenceRequest:
agent_id: str
action_type: str
amount: Optional[float] = None
target: Optional[str] = None

@dataclass
class CoherenceResponse:
action: str  # "EXECUTE" | "SILENCE" | "EMERGENCY_HALT"
psi: float
threshold: float
gap: float
limiting_plane: Optional[str]
trend: str
eta_seconds: Optional[float]
sentinel_score: float

class CausalSentinelClient:
def init(self, api_url: str, agent_id: str, api_key: str):
self.api_url = api_url
self.agent_id = agent_id
self.api_key = api_key

    async def evaluate_action(self, request: CoherenceRequest) -> CoherenceResponse:
        """Evaluate if action should execute based on coherence gate"""
        response = requests.post(
            f"{self.api_url}/api/v1/coherence/evaluate",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={
                "agent_id": self.agent_id,
                "action_type": request.action_type,
                "amount": request.amount,
                "target": request.target,
            }
        )
        data = response.json()
        return CoherenceResponse(**data)
    
    async def generate_zk_proof(self, credential_type: str) -> Dict[str, Any]:
        """Generate ZK proof for behavioral credential"""
        response = requests.post(
            f"{self.api_url}/api/v1/zk/generate",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={
                "agent_id": self.agent_id,
                "credential_type": credential_type,
            }
        )
        return response.json()
    
    async def get_moat_status(self) -> Dict[str, Any]:
        """Get current moat Λ(t) status"""
        response = requests.get(
            f"{self.api_url}/api/v1/moat/{self.agent_id}",
            headers={"Authorization": f"Bearer {self.api_key}"}
        )
        return response.json()
